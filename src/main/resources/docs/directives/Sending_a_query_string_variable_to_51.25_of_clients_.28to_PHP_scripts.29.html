[edit] Sending a query string variable to 51% of clients (to PHP scripts) 

http {
    split_clients "${remote_addr}" $additionalQsVariable {
	    51% "coolVariable=1";
	    49% "";
	}
 
    server {
        location ~ \.php$ {
            set $args "${query_string}&${additionalQsVariable}";
            [...]
        }
    }

Module: HttpSplitClientsModule
